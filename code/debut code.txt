import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import ast

# Charger le dataset
url = "https://raw.githubusercontent.com/vamshi121/TMDB-5000-Movie-Dataset/master/tmdb_5000_movies.csv"
movies = pd.read_csv(url)

# Parser la colonne genres (stock√©e comme JSON)
movies['genres'] = movies['genres'].apply(lambda x: [i['name'] for i in ast.literal_eval(x)] if pd.notnull(x) else [])
